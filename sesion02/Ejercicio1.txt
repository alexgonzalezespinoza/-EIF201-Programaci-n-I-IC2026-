#include "funciones.h"

double promedio(double numeros[]) {
    double suma = 0;

    for (int i = 0; i < 10; i++) {
        suma += numeros[i];
    }

    return suma / 10;
}

double mayor(double numeros[]) {
    double mayor = numeros[0];

    for (int i = 1; i < 10; i++) {
        if (numeros[i] > mayor) {
            mayor = numeros[i];
        }
    }

    return mayor;
}

double menor(double numeros[]) {
    double menor = numeros[0];

    for (int i = 1; i < 10; i++) {
        if (numeros[i] < menor) {
            menor = numeros[i];
        }
    }

    return menor;
}

double sumaElementos(double numeros[]) {
    double n = 0;

    for (int i = 0; i < 10; i++) {
        n += numeros[i];
    }

    return n;
}
