matriz.cpp
#include "matriz.h"
#include <iostream>
#include <iomanip>

using namespace std;

void leerMatriz(int matriz[Filas][Cols]) {
	for (int i = 0; i < Filas; i++) {
		for (int j = 0; j < Cols; j++) {
			cout << "Ingrese los datos " << endl; cin >> matriz[i][j];
		}
	}
}
void imprimir(int matriz[Filas][Cols]) {
	for (int i = 0; i < Filas; i++) {
		for (int j = 0; j < Cols; j++) {
			cout << setw(6) << matriz[i][j];
		}
		cout << endl;
	}
}
void sumaFilas(int matriz[Filas][Cols]) {

	for (int i = 0; i < Filas; i++) {
		int suma = 0;
		for (int j = 0; j < Cols; j++) {
			suma += matriz[i][j];
		}
		cout << "la suma de la fila es " << suma << endl;
	}

}
void sumaColum(int matriz[Filas][Cols]) {

	for (int j = 0; j < Cols; j++) {
		int suma = 0;
		for (int i = 0; i < Filas; i++) {
			suma += matriz[i][j];
		}
		cout << "la suma de la columna es " << suma << endl;
	}

}
void sumaDiagonal(int matriz[Filas][Cols]) {
	int suma = 0;
	for (int i = 0; i < Filas; i++) {
		suma += matriz[i][i];
	}
	cout << "la suma de la diagonal es " << suma << endl;
}
