#include "nombre.h"
#include <cctype>

string obtenerPrimerNombre(string nombre) {
    int pos = nombre.find(" ");
    return nombre.substr(0, pos);
}


string obtenerPrimerApellido(string nombre) {
    int pos = nombre.find(" ");
    int pos2 = nombre.find(" ", pos + 1);
    return nombre.substr(pos + 1, pos2 - pos - 1);
}


int contarVocales(string nombre) {
    int contador = 0;

    for (int i = 0; i < nombre.length(); i++) {
        char letra = tolower(nombre[i]);

        if (letra == 'a' || letra == 'e' || letra == 'i' || letra == 'o' || letra == 'u') {
            contador++;
        }
    }

    return contador;
}


string convertirMayusculas(string nombre) {

    for (int i = 0; i < nombre.length(); i++) {
        nombre[i] = toupper(nombre[i]);
    }

    return nombre;
}
